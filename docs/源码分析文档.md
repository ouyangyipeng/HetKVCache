```cpp
    // 阈值
    float hot_threshold_;
    float warm_threshold_;
    
    // 分数计算方法
    float computeFrequencyScore(uint64_t access_count, Timestamp current_time) const;
    float computeRecencyScore(Timestamp last_access, Timestamp current_time) const;
    float computeLayerImportanceScore(LayerId layer_id, uint32_t num_layers) const;
```

**核心算法实现**：

```cpp
float HeatEvaluator::computeHeatScore(BlockId block_id,
                                       const std::vector<AccessRecord>& history) const {
    if (history.empty()) return 0.0f;
    
    Timestamp current_time = getCurrentTime();
    
    // 1. 计算频率分数
    uint64_t access_count = history.size();
    float freq_score = computeFrequencyScore(access_count, current_time);
    
    // 2. 计算新近度分数
    Timestamp last_access = history.back().timestamp;
    float recency_score = computeRecencyScore(last_access, current_time);
    
    // 3. 计算层重要性分数（取平均）
    float layer_score = 0.0f;
    for (const auto& record : history) {
        layer_score += computeLayerImportanceScore(record.layer_id, num_layers_);
    }
    layer_score /= history.size();
    
    // 4. 加权综合
    return alpha_ * freq_score + beta_ * recency_score + gamma_ * layer_score;
}

float HeatEvaluator::computeFrequencyScore(uint64_t access_count,
                                           Timestamp current_time) const {
    // 指数衰减模型：频率越高，分数越高，但有上限
    // score = 1 - exp(-lambda * count)
    return 1.0f - std::exp(-lambda_ * static_cast<float>(access_count));
}

float HeatEvaluator::computeRecencyScore(Timestamp last_access,
                                         Timestamp current_time) const {
    // 时间衰减模型：越近访问，分数越高
    // score = exp(-delta_t / tau)
    constexpr float tau = 1e9f * 60.0f; // 60秒时间常数
    float delta_t = static_cast<float>(current_time - last_access);
    return std::exp(-delta_t / tau);
}

float HeatEvaluator::computeLayerImportanceScore(LayerId layer_id,
                                                 uint32_t num_layers) const {
    // Transformer层重要性：浅层更重要
    float relative_pos = static_cast<float>(layer_id) / num_layers;
    if (relative_pos < 0.33f) return 1.0f;      // 浅层
    else if (relative_pos < 0.67f) return 0.8f; // 中层
    else return 0.6f;                            // 深层
}
```

```cpp
class SequentialPatternDetector {
public:
    bool detect(const std::vector<BlockId>& sequence) {
        if (sequence.size() < MIN_PATTERN_LENGTH) return false;
        
        // 检查是否为顺序访问
        int sequential_count = 0;
        for (size_t i = 1; i < sequence.size(); ++i) {
            if (sequence[i] == sequence[i-1] + 1) {
                sequential_count++;
            }
        }
        
        float sequential_ratio = static_cast<float>(sequential_count) / 
                                 (sequence.size() - 1);
        return sequential_ratio > SEQUENTIAL_THRESHOLD;  // 0.8
    }
};

class StridedPatternDetector {
public:
    std::optional<int64_t> detectStride(const std::vector<BlockId>& sequence) {
        if (sequence.size() < MIN_PATTERN_LENGTH) return std::nullopt;
        
        // 计算步长
        std::unordered_map<int64_t, int> stride_counts;
        for (size_t i = 1; i < sequence.size(); ++i) {
            int64_t stride = static_cast<int64_t>(sequence[i]) - 
                            static_cast<int64_t>(sequence[i-1]);
            stride_counts[stride]++;
        }
        
        // 找出最频繁的步长
        auto max_it = std::max_element(stride_counts.begin(), stride_counts.end(),
            [](const auto& a, const auto& b) { return a.second < b.second; });
        
        float stride_ratio = static_cast<float>(max_it->second) / 
                             (sequence.size() - 1);
        if (stride_ratio > STRIDE_THRESHOLD) {  // 0.7
            return max_it->first;
        }
        return std::nullopt;
    }
};
```

**迁移任务结构**：

```cpp
struct MigrationTask {
    BlockId block_id;
    TierType source_tier;
    TierType target_tier;
    MigrationPriority priority;
    std::chrono::steady_clock::time_point submit_time;
    std::unique_ptr<std::promise<bool>> completion_promise;
};

struct MigrationTaskComparator {
    bool operator()(const MigrationTask& a, const MigrationTask& b) const {
        // 优先级高的排前面
        if (a.priority != b.priority) {
            return static_cast<int>(a.priority) < static_cast<int>(b.priority);
        }
        // 同优先级，提交时间早的排前面
        return a.submit_time > b.submit_time;
    }
};
```

**工作线程实现**：

```cpp
void MigrationEngine::workerThread() {
    while (running_.load()) {
        MigrationTask task;
        
        // 从队列获取任务
        {
            std::unique_lock<std::mutex> lock(queue_mutex_);
            queue_cv_.wait(lock, [this] {
                return !pending_queue_.empty() || !running_.load();
            });
            
            if (!running_.load()) break;
            
            task = pending_queue_.top();
            pending_queue_.pop();
        }
        
        // 执行迁移
        bool success = executeMigration(task);
        
        // 设置结果
        task.completion_promise->set_value(success);
    }
}

bool MigrationEngine::executeMigration(const MigrationTask& task) {
    // 1. 获取源和目标内存指针
    void* src_ptr = getBlockPointer(task.block_id, task.source_tier);
    void* dst_ptr = getBlockPointer(task.block_id, task.target_tier);
    
    if (!src_ptr || !dst_ptr) return false;
    
    // 2. 执行数据传输
    bool success = scheduler_->scheduleTransfer(
        src_ptr, dst_ptr, block_size_,
        getTierDevice(task.source_tier),
        getTierDevice(task.target_tier)
    );
    
    // 3. 更新元数据
    if (success) {
        updateBlockLocation(task.block_id, task.target_tier);
        updateStats(task.source_tier, task.target_tier);
    }
    
    return success;
}
```

**预取预测算法**：

```cpp
std::vector<PrefetchSuggestion> Prefetcher::generatePrefetchSuggestions(
        BlockId current_block) {
    
    std::vector<PrefetchSuggestion> suggestions;
    
    // 查找当前块的转移概率
    auto it = transition_counts_.find(current_block);
    if (it == transition_counts_.end()) {
        return suggestions;
    }
    
    // 计算转移概率并排序
    std::vector<std::pair<BlockId, float>> probabilities;
    uint64_t total_transitions = 0;
    
    for (const auto& [next_block, count] : it->second) {
        total_transitions += count;
    }
    
    for (const auto& [next_block, count] : it->second) {
        float prob = static_cast<float>(count) / total_transitions;
        if (prob >= prefetch_threshold_) {
            probabilities.push_back({next_block, prob});
        }
    }
    
    // 按概率排序
    std::sort(probabilities.begin(), probabilities.end(),
              [](const auto& a, const auto& b) { return a.second > b.second; });
    
    // 生成建议
    size_t count = std::min(probabilities.size(), max_prefetch_count_);
    for (size_t i = 0; i < count; ++i) {
        PrefetchSuggestion suggestion;
        suggestion.block_id = probabilities[i].first;
        suggestion.probability = probabilities[i].second;
        suggestion.priority = (i < 2) ? MigrationPriority::HIGH : 
                                        MigrationPriority::NORMAL;
        suggestions.push_back(suggestion);
    }
    
    return suggestions;
}
```

```cpp
BlockId PagedAllocator::allocateBlock() {
    std::lock_guard<std::mutex> lock(allocator_mutex_);
    
    // 尝试从VRAM分配（优先）
    BlockId block_id = allocateFromTier(TierType::HOT);
    if (block_id != INVALID_BLOCK_ID) {
        return block_id;
    }
    
    // VRAM不足，尝试从RAM分配
    block_id = allocateFromTier(TierType::WARM);
    if (block_id != INVALID_BLOCK_ID) {
        return block_id;
    }
    
    // RAM也不足，从SSD分配
    block_id = allocateFromTier(TierType::COLD);
    return block_id;
}

BlockId PagedAllocator::allocateFromTier(TierType tier) {
    auto& free_list = free_lists_[static_cast<size_t>(tier)];
    
    if (free_list.empty()) {
        return INVALID_BLOCK_ID;
    }
    
    // 从空闲列表头部取一个块
    BlockId block_id = free_list.back();
    free_list.pop_back();
    
    // 更新元数据
    auto& metadata = block_metadata_[block_id];
    metadata.state = BlockState::ALLOCATED;
    metadata.ref_count = 1;
    metadata.last_access = getCurrentTime();
    
    // 更新使用量
    tier_usages_[static_cast<size_t>(tier)]++;
    
    return block_id;
}
```

**迁移实现**：

```cpp
bool PagedAllocator::migrateBlock(BlockId block_id, TierType target_tier) {
    auto it = block_metadata_.find(block_id);
    if (it == block_metadata_.end()) return false;
    
    auto& metadata = it->second;
    TierType source_tier = metadata.tier;
    
    if (source_tier == target_tier) return true;  // 已经在目标层级
    
    // 检查目标层级是否有空间
    size_t target_idx = static_cast<size_t>(target_tier);
    if (tier_usages_[target_idx] >= tier_capacities_[target_idx]) {
        // 目标层级已满，需要驱逐
        if (!evictFromTier(target_tier)) {
            return false;
        }
    }
    
    // 获取源和目标指针
    void* src_ptr = getBlockPointer(block_id);
    void* dst_ptr = nullptr;
    
    size_t source_idx = static_cast<size_t>(source_tier);
    switch (target_tier) {
        case TierType::HOT:
            dst_ptr = static_cast<char*>(vram_pool_) + findFreeOffset(TierType::HOT);
            // CUDA拷贝：RAM/SSD -> VRAM
            cudaMemcpy(dst_ptr, src_ptr, config_.block_size, cudaMemcpyDefault);
            break;
            
        case TierType::WARM:
            dst_ptr = static_cast<char*>(ram_pool_) + findFreeOffset(TierType::WARM);
            if (source_tier == TierType::HOT) {
                // CUDA拷贝：VRAM -> RAM
                cudaMemcpy(dst_ptr, src_ptr, config_.block_size, cudaMemcpyDeviceToHost);
            } else {
                // SSD -> RAM
                pread(ssd_fd_, dst_ptr, config_.block_size, metadata.offset);
            }
            break;
            
        case TierType::COLD:
            size_t offset = findFreeOffset(TierType::COLD);
            if (source_tier == TierType::HOT) {
                // VRAM -> SSD
                char* temp = new char[config_.block_size];
                cudaMemcpy(temp, src_ptr, config_.block_size, cudaMemcpyDeviceToHost);
                pwrite(ssd_fd_, temp, config_.block_size, offset);
                delete[] temp;
            } else {
                // RAM -> SSD
                pwrite(ssd_fd_, src_ptr, config_.block_size, offset);
            }
            metadata.offset = offset;
            break;
    }
    
    // 更新元数据
    tier_usages_[source_idx]--;
    tier_usages_[target_idx]++;
    metadata.tier = target_tier;
    metadata.offset = /* 新偏移 */;
    
    return true;
}
```

```cpp
// 批量传输
cudaError_t batchMemcpy(const std::vector<MemcpyRequest>& requests,
                        cudaStream_t stream) {
    for (const auto& req : requests) {
        cudaError_t err = cudaMemcpyAsync(req.dst, req.src, req.size,
                                          req.kind, stream);
        if (err != cudaSuccess) return err;
    }
    return cudaSuccess;
}

// 带宽测试
float measureBandwidth(void* dst, void* src, size_t size,
                       cudaMemcpyKind kind, int iterations) {
    cudaEvent_t start, stop;
    cudaEventCreate(&start);
    cudaEventCreate(&stop);
    
    cudaStream_t stream;
    cudaStreamCreate(&stream);
    
    float total_time = 0.0f;
    
    for (int i = 0; i < iterations; ++i) {
        cudaEventRecord(start, stream);
        cudaMemcpyAsync(dst, src, size, kind, stream);
        cudaEventRecord(stop, stream);
        cudaEventSynchronize(stop);
        
        float elapsed;
        cudaEventElapsedTime(&elapsed, start, stop);
        total_time += elapsed;
    }
    
    cudaEventDestroy(start);
    cudaEventDestroy(stop);
    cudaStreamDestroy(stream);
    
    // 返回GB/s
    return (size / 1e9) / (total_time / (iterations * 1000));
}

} // namespace cuda
} // namespace hetkvcache
```

```cpp
// KV Cache合并内核
__global__ void mergeKVCacheKernel(
    const float* __restrict__ src_k,
    const float* __restrict__ src_v,
    float* __restrict__ dst_k,
    float* __restrict__ dst_v,
    size_t head_dim,
    size_t num_heads) {
    
    size_t idx = blockIdx.x * blockDim.x + threadIdx.x;
    size_t total = head_dim * num_heads;
    
    if (idx < total) {
        dst_k[idx] = src_k[idx];
        dst_v[idx] = src_v[idx];
    }
}

// 热度分数计算内核
__global__ void computeHeatScoreKernel(
    const uint64_t* __restrict__ access_counts,
    const uint64_t* __restrict__ last_access_times,
    float* __restrict__ heat_scores,
    size_t num_blocks,
    uint64_t current_time,
    float lambda) {
    
    size_t idx = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (idx < num_blocks) {
        // 频率分数
        float freq_score = 1.0f - expf(-lambda * access_counts[idx]);
        
        // 新近度分数
        float delta_t = static_cast<float>(current_time - last_access_times[idx]);
        float recency_score = expf(-delta_t / (1e9f * 60.0f)); // 60秒衰减
        
        // 综合分数
        heat_scores[idx] = 0.5f * freq_score + 0.5f * recency_score;
    }
}

// 块压缩内核（可选功能）
__global__ void compressBlockKernel(
    const float* __restrict__ input,
    half* __restrict__ output,
    size_t size) {
    
    size_t idx = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (idx < size) {
        output[idx] = __float2half(input[idx]);
    }
}
```

```cpp
class CUDAStreamManager {
public:
    CUDAStreamManager(size_t num_streams = 4) {
        streams_.resize(num_streams);
        for (auto& stream : streams_) {
            cudaStreamCreate(&stream);
        }
        current_stream_ = 0;
    }
    
    ~CUDAStreamManager() {
        for (auto& stream : streams_) {
            cudaStreamDestroy(stream);
        }
    }
    
    cudaStream_t getNextStream() {
        size_t idx = current_stream_.fetch_add(1) % streams_.size();
        return streams_[idx];
    }
    
    void synchronizeAll() {
        for (auto& stream : streams_) {
            cudaStreamSynchronize(stream);
        }
    }
    
private:

    std::atomic<size_t> current_stream_;
};
```

```cpp
            case LogLevel::WARN:  return "WARN";
```

```cpp
class Timer {
public:
    Timer() : start_(std::chrono::high_resolution_clock::now()) {}
    
    void reset() {
        start_ = std::chrono::high_resolution_clock::now();
    }
    
    double elapsedNs() const {
        auto now = std::chrono::high_resolution_clock::now();
        return std::chrono::duration<double, std::nano>(now - start_).count();
    }
    
    double elapsedUs() const {
        return elapsedNs() / 1000.0;
    }
    
    double elapsedMs() const {
        return elapsedNs() / 1000000.0;
    }
    
    double elapsedSec() const {
        return elapsedNs() / 1000000000.0;
    }
    
private:

};
```
